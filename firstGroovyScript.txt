class Customer {
    def name
    def country
    
    Customer(def aName, def aCountry) {
        name = aName
        country = aCountry
    }
    
    def getName() {
        return name
    }
    
    void setName(def aName) {
        name = aName
    }
    
    def getCountry() {
        return country
    }
    
    void setCountry(def aCountry) {
        country = aCountry
    }
}

Customer customer = new Customer("Natalia", "GB");
println customer.getName();

println customer1.getName();
Customer customer2 = new Customer("Maria", "UK");
